

import 'package:dio/dio.dart';
import 'package:shared_preferences/shared_preferences.dart';

class AuthService {
  final Dio _dio = Dio();
  
  // تذكر: استخدم IP جهازك في الشبكة بدلاً من 127.0.0.1 للموبايل الحقيقي
  final String baseUrl = "https://gps-liver-null-titles.trycloudflare.com/api/auth"; 

  Future<void> register(String name, String email, String password) async {
    try {
      Response response = await _dio.post(
        "$baseUrl/register",
        data: {
          "name": name,
          "email": email,
          "password": password,
          "password_confirmation": password, // تأكد من إرسال هذا كما فعلنا في Postman
        },
        options: Options(headers: {"Accept": "application/json"}),
      );

      if (response.statusCode == 200) {
        String token = response.data['token'];
        
        // حفظ التوكن في ذاكرة الموبايل
        final prefs = await SharedPreferences.getInstance();
        await prefs.setString('user_token', token);
        
        print("Registration Successful! Token saved.");
      }
    } on DioException catch (e) {
      print("Error: ${e.response?.data['message']}");
    }
  }
}
